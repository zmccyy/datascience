from sklearn.model_selection import GridSearchCV, cross_val_score
from sklearn.pipeline import Pipeline

# 创建更完整的机器学习流程
def create_ml_pipeline():
    """创建包含预处理和分类器的管道"""
    numerical_columns = ['satisfaction_level', 'last_evaluation', 'number_project',
                         'average_montly_hours', 'time_spend_company']
    
    # 创建预处理转换器
    preprocessor = ColumnTransformer(
        transformers=[
            ('num', StandardScaler(), numerical_columns),
            # 分类变量已经进行了独热编码，不需要额外处理
        ],
        remainder='passthrough'  # 保留其他列
    )
    
    # 创建管道
    pipeline = Pipeline([
        ('preprocessor', preprocessor),
        ('classifier', RandomForestClassifier(random_state=42))
    ])
    
    return pipeline

# 使用网格搜索进行超参数优化
def optimize_hyperparameters(X, y):
    pipeline = create_ml_pipeline()
    
    param_grid = {
        'classifier__n_estimators': [50, 100, 200],
        'classifier__max_depth': [None, 10, 20, 30],
        'classifier__min_samples_split': [2, 5, 10],
        'classifier__min_samples_leaf': [1, 2, 4]
    }
    
    grid_search = GridSearchCV(
        pipeline, param_grid, cv=5, scoring='accuracy', 
        n_jobs=-1, verbose=1
    )
    
    grid_search.fit(X, y)
    
    print("最佳参数:", grid_search.best_params_)
    print("最佳交叉验证分数:", grid_search.best_score_)
    
    return grid_search.best_estimator_

# 使用交叉验证评估模型
def evaluate_model(model, X, y):
    cv_scores = cross_val_score(model, X, y, cv=5, scoring='accuracy')
    print(f"交叉验证准确率: {cv_scores.mean():.4f} (±{cv_scores.std()*2:.4f})")
    
    # 训练最终模型
    model.fit(X, y)
    return model

from imblearn.over_sampling import SMOTE
from imblearn.under_sampling import RandomUnderSampler

# 处理类别不平衡
def handle_imbalance(X, y, strategy='smote'):
    if strategy == 'smote':
        sampler = SMOTE(random_state=42)
    elif strategy == 'undersample':
        sampler = RandomUnderSampler(random_state=42)
    else:
        return X, y
    
    X_res, y_res = sampler.fit_resample(X, y)
    print(f"重采样后类别分布: {pd.Series(y_res).value_counts().to_dict()}")
    return X_res, y_res

from sklearn.metrics import classification_report, confusion_matrix, roc_curve, auc, precision_recall_curve

def comprehensive_evaluation(model, X_test, y_test):
    y_pred = model.predict(X_test)
    y_prob = model.predict_proba(X_test)[:, 1]
    
    # 分类报告
    print("分类报告:")

    ax1.set_title('Confusion Matrix')
    
    # ROC曲线
    fpr, tpr, _ = roc_curve(y_test, y_prob)
    roc_auc = auc(fpr, tpr)
    
    ax2.plot(fpr, tpr, color='darkorange', lw=2, label=f'ROC curve (AUC = {roc_auc:.2f})')

def plot_feature_importance(model, feature_names, top_n=15, save_path='feature_importance.svg'):
    """绘制特征重要性图"""
    if hasattr(model, 'feature_importances_'):
        importances = model.feature_importances_
    elif hasattr(model, 'coef_'):
        importances = np.abs(model.coef_[0])
    else:
        print("模型没有特征重要性属性")
        return
    
    # 创建特征重要性DataFrame
